\newcommand{\ourhead}[5] {
\chead{
  \ifnum\thepage=1
  \else
      \makebox[\textwidth]{
      \Ifstr{#1}{}{}{#1: #2 \hspace{0.7cm}}
      #3
      \Ifstr{#4}{}{}{~\nr: #4}
      \Ifstr{#5}{}{}{\vom ~ #5}

      }
  \fi
}
}
